318void HACCGPM::CosmoClass::growths(float a, float* y, float* dydx) {
319 float H;
320 H = sqrt(Omega_cb/pow(a,3.0f)
321 + (1.0 + f_nu_massless)*Omega_r/pow(a,4.0f)
322 + Omega_nu_massive(a)
323 + (1.0 - Omega_m - (1.0+f_nu_massless)*Omega_r)
